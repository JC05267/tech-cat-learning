## Pandas
```python
import pandas as pd

df = pd.read_csv('file_path.csv')

# preview
df.head(n=10) # default 5
df.tail()

df.sample(5) # default 1

# slices - better to slice on read via `usecols`
df[['col1', 'col2', 'col3',]]

# Date casting - can do on read via 'parse_dates'
df['date'] = pd.to_datetime(df['date'])
df['month'] = df['date'].dt.month
df['year'] = df['year'].dt.year

# time series - can do on read via `index_col`
df.set_index('date')

# data cleaning
df['price_cleaned'] = df['price'].str.replace('<thing to replace>', "<token>").replace("<other thing>", "<other token>")

clean = lambda x: x.str.replace('<to replace>', '<token>')
c_df = df[['col1','col2']].apply(clean, axis=1)

# plots
df.value_counts('city').plot(kind='bar')

# group by
df.groupby(by=['col1', 'col2']).sum()
df.groupby(by=['col1', 'col2']).count()

# sort
df.sort_values(by=['col1'])

# secret
# using `storage options` you can pass credentials to pandas for easy loading
```
